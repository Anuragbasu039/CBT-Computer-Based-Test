    public void check()
    {
    String resans="";

    }
    else if(opt2.isSelected())
    {

    }
     else if(opt3.isSelected())
    {
    resans=opt3.getText();
    }

    {
    resans=opt4.getText();
    }
    if(resans.equals(ans))
    {
    marks=marks+1;
    }
    }

    private void btnSubmitActionPerformed(java.awt.event.ActionEvent evt) {                                          
        // TODO add your handling code here:
        rs=dm.select("select* from student where rollno='"+rollno+"'");
        try
        {
        if(rs.next())
        {
        String name=rs.getString("name");
        String branch=rs.getString("branch");
        String year=rs.getString("year");
        String query="insert into result(rollno,name,branch,year,marks)values('"+rollno+"','"+name+"','"+branch+"','"+year+"','"+marks+"')";
        boolean res=dm.insertUpdateDelete(query);
        if(res==true)
                {
                  JOptionPane.showMessageDialog(this,"Success your marks="+marks);
                }
        else
        {
          JOptionPane.showMessageDialog(this,"Your marks Not Submitted="+marks);
        }
        this.setVisible(false);
        new MainForm().setVisible(true);
        }
        }
         catch(SQLException ex)
        {
        ex.printStackTrace();
        }
        //JOptionPane.showMessageDialog(this,"You secure marks= "+marks);
        
    }
